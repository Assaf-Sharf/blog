```javascript
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
const input = fs.readFileSync(filePath).toString().split(' ').map(item => +item);

function solution(marks) {
    if (marks >= 90) {
        console.log('A');
    } else if (marks >= 80) {
        console.log('B');
    } else if (marks >= 70) {
        console.log('C');
    } else if (marks >= 60) {
        console.log('D');
    } else {
        console.log('F');
    }
};

solution(input);
```

아래처럼 return을 사용해서 할 수도 있다

```javascript
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
const input = fs.readFileSync(filePath).toString().split(' ').map(item => +item);

function solution(marks) {
    if (marks >= 90) {
        grade = 'A';
    } else if (marks >= 80) {
        grade = 'B';
    } else if (marks >= 70) {
        grade = 'C';
    } else if (marks >= 60) {
        grade = 'D';
    } else {
        grade = 'F';
    }
    return grade;
};

console.log(solution(input));
```
