def quick_sort(arr):
    if len(arr) <= 1:
        return arr  
    
    pivot = arr[len(arr) // 2]  # Pivot MEDIAN
    left = []
    right = []
    equal = []
    
    # 피벗을 기준으로 왼쪽, 오른쪽, 같은 값 리스트로 분할
    for num in arr:
        if num < pivot:
            left.append(num)
        elif num > pivot:
            right.append(num)
        else:
            equal.append(num)
    
    # 왼쪽, 오른쪽 리스트를 재귀적으로 정렬하고, 각각을 합쳐서 반환
    return quick_sort(left) + equal + quick_sort(right)
