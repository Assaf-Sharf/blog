void hanoi(int n, char from, char temp, char to) {
	if (n == 1) {
		printf(" 원판1 : 탑 %c에서 탑 %c로 이동\n", from, to);
	}
	else {
		hanoi(n - 1, from, to, temp);
		printf(" 원판%d : 탑 %c에서 탑 %c로 이동\n", n, from, to);
		hanoi(n - 1, temp, from, to);
	}
}

int GCD(int a, int b) {

	int tmp;

	if (a < b) {
		tmp = a;
		a = b;
		b = tmp;
	}

	if (b == 0) {
		return a;
	}
	else
		return GCD(b, a % b);

}

int LCM(int a, int b) {
	return a * b / GCD(a, b);
}
