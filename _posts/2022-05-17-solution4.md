```python
from sys import stdin
n,m = int(input())

status = [[0]*n for i in range(n+1)]

for _ in range(m):
    a,b = map(int,input().split())
    status[a].append(b)
    status[b].append(a)

cnt = 0
visited = [0]*(n+1)

def dfs(start):
    global cnt
    visited[start] = 1
    for i in status[start]:
        if visited[i]==0:
            dfs(i)
            cnt +=1

print(cnt)
```
