```
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
const input = fs.readFileSync(filePath).toString().split(' ').map(item => +item); // 문자열로 바꿔주고 > 공백을 기준으로 split해주고(split을 먼저 해줘야 배열의 형태로 바뀐다) > 그 다음에 map을 이용해서 배열을 돌면서 모두 숫자로 바꿔주기

function solution(A,B) {
    console.log(A-B);
};

solution(input[0], input[1]);
```

위의 코드가 어렵다면 console.log로 찍어서 확인해가면서 할 수도 있다 (코드는 더 길어지지만 풀어서 썼으므로 이해하기는 더 쉽다. 계산하는데 걸리는 시간도 더 짧다.)
```
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
let input = fs.readFileSync(filePath).toString(); // let으로 바꿔주기
console.log(input); // 3 2

input = input.split('\n'); // '\n'을 기준으로 split을 먼저 해준다 (split을 해주면 배열이 된다)
console.log(input); // ['3 2']

input = input[0]; // 배열의 0번쨰를 꺼내주자
console.log(input); // 3 2 (vs code 터미널에 찍어보면 회색으로 뜬다. 문자열은 회색, 숫자는 노란색)

input = input.split(' ').map((item) => +item); // 이 3 2를 split()을 써서 공백을 기준으로 나눠준다. ['3','2'] 이렇게 문자열로 잘 나뉘어짐. 자, 우리가 풀어야하는 문제는 더하기 문제이므로 숫자로 바꿔줘야한다. map을 이용해서 배열의 원소들을 전부 돌면서 숫자로 바꿔준다. 'Number(item)' 이렇게 해도 되고 '+item'이렇게도 가능
console.log(input); // [3, 2]

function solution(A,B) {
   console.log(A - B);  // 1
};

solution(input[0], input[1]);
