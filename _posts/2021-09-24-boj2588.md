<https://www.acmicpc.net/problem/2588>

```javascript
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
let input = fs.readFileSync(filePath).toString().split('\n').map(item => +item); // 입력을 받았으니 split을 이용해서 입력 받은 값을 쪼개줌(여기서는 '\n'을 기준으로 쪼갰다)과 동시에 배열의 형태로 바꿔준다 > 그 다음 map을 이용하여 배열을 쭉 돌면서 숫자로 바꿔준다

console.log(input); // [ 472, 385 ] 배열의 형태로 잘 쪼개졌다

function solution(A, B) {
    console.log(A*5); // 2360
    console.log(A*8); // 3776
    console.log(A*3); // 1416
    console.log(A*B); // 181720
};

solution(input[0], input[1]);
```

문제가 원하는 출력값이 잘 나온다. <br/> 
하지만 B에 385가 아닌 다른 숫자를 넣으면 활용을 못하게 된다.<br/> 
한마디로 활용성이 떨어진다.<br/>
그러므로 아래처럼 변경 :

```javascript
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
let input = fs.readFileSync(filePath).toString().split('\n').map(item => +item);

console.log(input); // [ 472, 385 ]

function solution(A, B) {
    const StringB = String(B); // B에 담기는건 숫자이므로 문자열로 변환해준다. 문지열로 변환해준다는 말은 곧 배열을 쓸 수 있다는 소리다
    console.log(typeof(StringB)); // 잘 변환됐는지 확인 
    console.log(StringB); // 385 (vs code 터미널에 찍어보면 회색으로 잘뜬다. 회색은 문자열. 노란색은 숫자.)

    console.log(A*StringB[2]); // 5
    console.log(A*StringB[1]); // 8
    console.log(A*StringB[0]); // 3
    console.log(A*B);
};

solution(input[0], input[1]);
```

```javascript
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
let input = fs.readFileSync(filePath).toString().split('\n').map(item => +item);

console.log(input); // [ 472, 385 ]

function solution(A, B) {
    const StringB = String(B); // B에 담기는건 숫자이므로 문자열로 변환해준다. 문지열로 변환해준다는 말은 곧 배열을 쓸 수 있다는 소리다
    const ones = StringB[2]; // 5
    const tens = StringB[1]; // 8
    const hundreds = StringB[0]; // 3
  
    console.log(A*ones);
    console.log(A*tens);
    console.log(A*hundreds);
    console.log(A*B);
};

solution(input[0], input[1]);

```
