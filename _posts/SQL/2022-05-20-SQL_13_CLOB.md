> ### CLOB 대용량 데이터타입

~~~sql
-- CLOB 대용량 데이터타입 : 성능이슈. 자료확인 권장

CREATE TABLE ex3_5_2(
    NAMES CLOB -- 4GB 대용량데이터타입
);

INSERT INTO ex3_5_2 VALUES('홍길');
INSERT INTO ex3_5_2 VALUES('홍길동');
INSERT INTO ex3_5_2 VALUES('홍길용');
INSERT INTO ex3_5_2 VALUES('홍길상');
INSERT INTO ex3_5_2 VALUES('홍길동상');
INSERT INTO ex3_5_2 VALUES('1홍길동상');

SELECT * FROM ex3_5_2;

SELECT * FROM ex3_5_2
WHERE NAMES LIKE '홍길%';

-- ORDER BY 정렬 : 지원안됨
SELECT * FROM ex3_5
ORDER BY NAMES;

SELECT LENGTH   (NAMES) FROM ex3_5_2; -- 문자열 길이. 가능
SELECT LENGTHB  (NAMES) FROM ex3_5_2; -- 데이터 크기. 불가능

-- 이름이 '홍길' 이라는 데이터를 조회하라
SELECT * FROM ex3_5_2
WHERE NAMES LIKE '%홍길%';
~~~
