> ### UPDATE 문

- 테이블의 데이터를 수정할 때 사용하는 명령어

~~~sql
-- 사용법
UPDATE 테이블 
SET 칼럼 = 데이터, 칼럼 IS NULL, 칼럼 IS NOT NULL
WHERE 칼럼 = NAME;
~~~



~~~sql
-- WHERE 조건식 제외하면, 테이블의 모든 데이터를 대상으로 변경
UPDATE ex3_1
SET COL2 = 50;

UPDATE ex3_1
SET COL2 = 10, COL3 = '2022-05-20'
WHERE COL1 = 'GHI';

-- COL3컬럼이 NULL 인 데이터를 조회
-- 잘못된 예
SELECT *
FROM ex3_1
WHERE COL3 = ''; -- '' 오라클의 NULL의 의미이지만, 조건식에서는 NULL의미로 사용X. 데이터삽입시는 ''가 NULL의미

-- COL3컬럼에 데이터가 NULL인 데이터
SELECT *
FROM ex3_1
WHERE COL3 IS NULL;

-- 위의 것의 반대의미. COL3컬럼에 데이터가 존재하는 것
SELECT *
FROM ex3_1
WHERE COL3 IS NOT NULL;
~~~
