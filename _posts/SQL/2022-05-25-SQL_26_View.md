> ### 뷰

- SELECT문을 따로 저장해서 쉽게 사용하기 위한 용도
- CREATE OR REPLACE VIEW [스키마.] 뷰명 AS
- SELECT 문장;

~~~sql
-- 사원번호, 사원이름, 부서ID, 부서명을 출력하라
SELECT EMPLOYEE_ID, EMP_NAME, E.DEPARTMENT_ID, DEPARTMENT_NAME
FROM EMPLOYEES E INNER JOIN DEPARTMENTS D
ON E.DEPARTMENT_ID = D.DEPARTMENT_ID;

-- 뷰 생성(CREATE), 뷰 수정(REPLACE)
CREATE OR REPLACE VIEW VW_EMP_DEPT
AS
SELECT EMPLOYEE_ID, EMP_NAME, E.DEPARTMENT_ID, DEPARTMENT_NAME
FROM EMPLOYEES E INNER JOIN DEPARTMENTS D
ON E.DEPARTMENT_ID = D.DEPARTMENT_ID;

-- 뷰 삭제
DROP VIEW VW_EMP_DEPT;

-- 뷰 사용
-- SELECT * FROM 테이블명 OR 뷰명
SELECT * FROM VW_EMP_DEPT;
~~~
