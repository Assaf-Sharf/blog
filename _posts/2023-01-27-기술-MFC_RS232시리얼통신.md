## 참고 CString Spilt 기능
```cpp
int splitString(CString str, CString var, CStringArray& strs)
{
	int count = 0;

	CString tempStr = str;

	int length = str.GetLength();

	while (length)
	{
		int find = tempStr.Find(var);
		if (find != -1)
		{
			CString temp = tempStr.Left(find);
			strs.Add(temp);
			int varLen = _tcslen(var);
			tempStr = tempStr.Mid(find + varLen);
		}
		else
		{
			strs.Add(tempStr);
			length = 0;
		}
		count++;

	}
	return count;
}

```
