#include <string>
#include <vector>
#include <iostream>
#include <math.h>

using namespace std;

float get_distance(int a_x, int a_y, int b_x, int b_y)
{
    return sqrt(pow(a_x-b_x,2)+pow(a_y-b_y,2));
}

string solution(vector<int> numbers, string hand) {
    string answer = "";
    int location_left=10;
    int location_right=12;
    int mid_x=0, mid_y=0, l_x=0, l_y=0, r_x=0, r_y=0;

    for(int i=0; i< numbers.size(); i++)
    {
        if(numbers[i]==0)
            numbers[i]=11;
        switch(numbers[i])
        {
            case 1: case 4: case 7:
            location_left=numbers[i];
            answer.push_back('L');
            break;
            case 3: case 6: case 9:
            location_right=numbers[i];
            answer.push_back('R');
            break;
            default:
            mid_x=(numbers[i]-1)/3;
            mid_y=((numbers[i]%3)+2)%3;
            l_x=(location_left-1)/3;
            l_y=((location_left%3)+2)%3;
            r_x=(location_right-1)/3;
            r_y=((location_right%3)+2)%3;
            int mid_to_left=get_distance(l_x,l_y,mid_x,mid_y);
            int mid_to_right=get_distance(r_x,r_y,mid_x,mid_y);
            if( mid_to_left == mid_to_right)
            {
                if(hand=="left")
                {
                    location_left=numbers[i];
                    answer.push_back('L');
                    break;
                }
                else
                {
                    location_right=numbers[i];
                    answer.push_back('R');
                    break;
                }
            }
            else
            {
                if(mid_to_left<mid_to_right) //left가 더 거리 짧음
                {
                    location_left=numbers[i];
                    answer.push_back('L');
                    break;
                }
                else
                {
                    location_right=numbers[i];
                    answer.push_back('R');
                    break;
                }
            }    
            
        }
    }
    // for(numbers 길이 만큼)
    // numbers[i]가 왼쪽이면 answer왼손 추가+왼손 움직이기
    // numbers[i]가 오른쪽이면 answer오른손 추가+오른손 움직이기
    // else(중간이면) 거리 계산, 거리 같으면 hand로 따지기
    
    return answer;
}
int main()
{
    vector<int> numbers={1, 3, 4, 5, 8, 2, 1, 4, 5, 9, 5};
    //                   L  R  L   L  L R  L  L  R  R  L
    string hand="right";
    string answer=solution(numbers, hand);
    
    cout<<"answer = "<<answer;

}
