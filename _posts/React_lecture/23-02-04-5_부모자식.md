# 부모자식 컴포넌트

: 컴포넌트 안에 컴포넌트를 넣을 수 있다(부모자식관계)

```react
import react from "react";
import "App.css";
// 자식
function Child() {
  return <div>나는 아들입니다.</div>;
}
//부모
function App() {
  return (
    <div>
      <Child />
      <Child />
     <div>나는 아들입니다.</div>
      <div>나는 아들입니다.</div>
    </div>
  );
}

//나는 아들입니다 4번 출력 
//<Child /> 4번 받은것과 동일함

//이런 중복 로직땜시 컴포넌트화(리팩토링)가필요
export default App;

```

```react
import React from "react";

function Son() {
  return <div>나는 자식</div>;
}
function Father() {
  return (
    <div>
      <Son />
      <div>나는 아빠</div>;
    </div>
  );
}
function GrandF() {
  return (
    <div>
      <Father />
      <div>나는 할배</div>;
    </div>
  );
}

function App() {
  return <GrandF />;
}

export default App;
```
