# 9. Props추출

```react
// Child.jsx

import React from "react";

function Child({ age, name, children }) { //구조분해할당해서 명시적으로
  console.log(age);
  console.log(name);
  console.log(children);
  return <div>Child</div>;
}

Child.defaultProps = {
  age: "기본나이",
}; //age 변수가 선언이안되었을 때 기본값
export default Child;
```
