<https://www.acmicpc.net/problem/2884>

```javascript
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
const input = fs.readFileSync(filePath).toString().split(' ').map(item => +item);

function solution(H, M) {
    if (M >= 45) { // 분침이 45분 이상이면 단순히 45분만큼 빼주면 된다 (예. 6:45 => 6:00)
            M =  M -45;
            console.log(H, M);
    } else if (M < 45) { // 문제는 분침이 45분 이하일 때 (예. 6:40 => 5:55)
        if (H > 0) { // 시침이 1~23시 사이라면
            M = 60 - (45-M);   // 분침은 '60분 - (45분 - M)' 이런식으로 45분을 빼주면 된다
            H = H-1; // 시침은 단순히 1시간 빼주면 된다
            console.log(H, M);
        } else {  // 시침이 24시 정각일 때는 (00:00시)
            M = 60 - (45-M); // 분침은 위와 똑같이 
            H = 23; // 시침은 한 시간을 빼주면 23시가 되니까 단순하게 그냥 명시해줬다
            console.log(H, M);
        }
    };
};
solution(input[0], input[1]);
```

더 짧은 코드 도전
```javascript
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
let input = fs.readFileSync(filePath).toString().split(' ').map(item => +item);

function solution(H, M) {
    M -= 45; 
    if (M < 0) {
        M += 60;
        H -= 1;
    }
    if (H < 0) {
        H = 23;
    }
    console.log(H, M);
}

solution(input[0], input[1]);
```
