```python
import sys

input=sys.stdin.readline
meetings=[]
for _ in range(int(input())):
  start, end = map(int,input().split())
  meetings.append((end,start)) #리스트 (start,end) 튜플 형태로 넣음

meetings.sort() #정렬
t=0
ans=0
for end, start in meetings:
  if t<= start:
    ans+=1
    t=end #종료시간

print(ans)
```
