### 코드

#### ver(1) - JOIN 사용

```sql
-- ANIMAL_INS 테이블과 ANIMAL_OUTS 테이블을 ANIMAL_ID 컬럼을 기준으로 조인합니다.
-- 조인 결과는 ANIMAL_ID가 같은 행만 매칭됩니다.
SELECT A.ANIMAL_ID, B.NAME
FROM ANIMAL_INS AS A
JOIN ANIMAL_OUTS AS B
ON A.ANIMAL_ID = B.ANIMAL_ID
-- ANIMAL_INS 테이블에서 가져온 동물의 입양일자가
-- ANIMAL_OUTS 테이블에서 가져온 동물의 입양일자보다 더 최근인 경우에만 선택합니다.
WHERE A.DATETIME > B.DATETIME
-- 선택된 행들을 입양일자 기준으로 오름차순 정렬합니다.
ORDER BY A.DATETIME
```
