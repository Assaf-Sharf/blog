<h3>[style.css]</h3>

```css
  .class-list {
    display: flex;
    list-style: none;
    padding: 0;
    overflow: none; /* overflow 를 hidden 으로 설정하면, transform 시 이미지 짤림 */
    margin-top: 0;
  }

  .class-card {
    width: 240px; 
    min-width: 240px;
    /* box-shadow: 블럭에 그림자를 적용하는 CSS 기능
       https://developer.mozilla.org/ko/docs/Web/CSS/box-shadow 
       offset-x | offset-y | blur-radius | spread-radius | color */
    box-shadow: 10px 10px 10px -5px rgba(25, 42, 70, 0.2);  
    transition: 0.3s;
    border-radius: 8px;
    margin: 10px 10px;
    touch-action: pan-x;
  }
  
  .class-card:hover {
    box-shadow: -20px 20px 28px -10px rgba(25, 42, 70, 0.13);
  }
  
  .class-image {
    width: 100%;
    border-radius: 8px 8px 0 0;
    height: 150px;
  }
  
  .class-container {
    padding: 0.5em 1em 1em 1em;
  }
  
  .class-container-skill {
    padding: 0;
    display: flex;
    justify-content: space-between;
    align-content: center;
    margin: 0 0 1em 0;
  }
  
  .class-container-skill-text {
    font-size: 0.9em;
    color: #949393;
    padding: 0.2em 0;
  }
  
  .class-container-skill-level {
    font-size: 0.9em;
    color: white;
    background-color: #0d47a1;
    padding: 0.2em;
  }
  
  img {
    border-radius: 8px 8px 0 0;
    height: 150px;
  }
  
  .class-maintitle {
    font-weight: bold;
    margin-bottom: 0.5em;
    color: #545e6f;
    width: 100%;
    white-space: nowrap;
    overflow: hidden; /* "visible" 외의 값으로 설정 */
    text-overflow: ellipsis;
  }
  
  .class-subtitle {
    font-size: 0.9rem;
    margin: 0;
    color: #545e6f;
    line-height: 1.5em;
    word-wrap: break-word;
    word-break: keep-all;
```
