```php

<?php 
$names = array('Brad', 'John', 'Jane', 'kim') ;
$count = 0;
while($count < count($names)) {
    echo"<li>Hi, my name is $names[$count]</li>";
    $count++;
}
?>

<p>Hi, my name is <?php echo $names[0]; ?>.</p>
<p>Hi, my name is <?php echo $names[1]; ?>.</p>
<p>Hi, my name is <?php echo $names[2]; ?>.</p>
```
