<https://www.acmicpc.net/problem/10430>

```javascript
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
const input = fs.readFileSync(filePath).toString().split(' ').map(item => +item); // 입력을 받았으면 split()을 써서 원하는 형태로 입력값을 쪼개(?)줌과 동시에 '배열'로 만들어주는 것이 1순위다 > 계산을 하려면 숫자로 변환해줘야 하므로 map을 이용하여 배열을 쭉 돌면서 숫자로 변환한다

function solution(A, B, C) {
   console.log((A+B)%C);  // 문제에서 출력하라는데로 복붙
   console.log(((A%C) + (B%C))%C);
   console.log((A*B)%C);
   console.log(((A%C) * (B%C))%C);
};

solution(input[0], input[1], input[2]);
```
