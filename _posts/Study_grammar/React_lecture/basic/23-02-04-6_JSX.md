import React from "react"; //rfc, rfce로 시작

export default function App() {
  const number = 11;
  const pTageStyle = {
    color: "blue",
  };
  return (
    // 리액트에서는 반드시 className까지 써야함
    <div className="test-class">
      <p> 안녕하세요 리액트입니다.</p>

        style={

          // color: "red",
        }
      >
        {number > 10 ? number + "은 10보다 크다" : number + "는 10보다 작다"}
      </p>
    </div>
  );
}
