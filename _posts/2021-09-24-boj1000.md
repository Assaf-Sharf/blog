<https://www.acmicpc.net/problem/1000>

```
const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
let input = fs.readFileSync(filePath).toString().split('\n'); // let으로 바꿔주기

input = input[0]; // 위에서 split()을 했기 때문에 배열이다. 배열의 0번쨰부터 꺼내줘야한다
// console.log(input); // console.log 찍어보면 문자열 1 2 가 나온다 (VS Code 터미널에 찍어보면 회색으로 나온다. 문자열은 회색, 숫자는 노란색)

input = input.split(' ').map((item) => +item); // 이 1 2를 split()을 써서 공백을 기준으로 나눠준다. ['1','2'] 이렇게 문자열로 잘 나뉘어짐. 자, 우리가 풀어야하는 문제는 더하기 문제이므로 숫자로 바꿔줘야한다. map을 이용해서 배열의 원소들을 다 돌면서 숫자로 바꿔준다. 'Number(item)' 이렇게 해도 되고 '+item'이렇게도 가능
// console.log(input); // [1, 2]

function solution(A,B) {
   console.log(A + B);
};

solution(input[0], input[1]);
```
