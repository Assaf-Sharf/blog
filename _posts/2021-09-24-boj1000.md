const fs = require('fs');
const filePath = process.platform === 'linux' ? '/dev/stdin' : './input.txt';
let input = fs.readFileSync(filePath).toString().split('\n'); // let으로 바꿔주기. split()을 했기 때문에 배열임. 배열의 0번쨰부터 꺼내줄꺼야

input = input[0];
console.log(input); // console.log 찍어보면 문자열 1 2 나옴 (VS Code 터미널에 찍어보면 문자는 회색으로 나옴. 숫자면 노란색)

input = input.split(' ').map((item) => +item); // 이번엔 1 2를 공백으로 나눠줄꺼야. ['1','2'] 이렇게 문자열로 잘 나뉘어짐. 우리가 푸는 문제는 더하기를 원하므로 숫자로 바꿔줘야지. map을 이용해서 배열의 원소들을 다 돌면서 숫자로 바꿔줄꺼임. 'Number(item)' 이렇게 해도 되고 '+item'이렇게도 가능
console.log(input); // [1, 2]

solution(input[0], input[1]);
