#include <stdio.h>

int main(void)
{
   int num[100] = {0};
   int i, count, N;
   int x=0;

   printf("몇 개의 정수를 출력할까요: ");
   scanf("%d", &N);

   printf("정수 입력: ");
   for(i=0; i<N; i++)
   {
      scanf("%d", &num[i]);
   }

   printf("어떤 숫자를 세시겠습니까? ");
   scanf("%d", &count);

   for(i=0; i<N; i++)
   {
      if(count == num[i])
      {
         x += 1;
      }
   }

   printf("%d\n", x);

   return 0;
}
