## 데모 페이지

심볼 인식

<label for="nameInput">이름을 입력하세요:</label>
<input type="text" id="nameInput">
<button id="submitButton">확인 버튼</button>

<p id="welcomeMessage"></p>
<script>
  // 사용자 이름을 입력받아 환영 메시지를 표시하는 함수
  function displayWelcomeMessage() {
      // 입력된 이름을 가져오기
      var name = document.getElementById("nameInput").value;
      
      // 환영 메시지 작성
      var welcomeMessage = "안녕하세요, " + name + "님! 자바스크립트 세계에 오신 것을 환영합니다!";
      
      // 환영 메시지를 화면에 표시
      document.getElementById("welcomeMessage").textContent = welcomeMessage;
  }

  // 버튼 클릭 이벤트에 displayWelcomeMessage 함수 연결
  document.getElementById("submitButton").addEventListener("click", displayWelcomeMessage);
</script>
