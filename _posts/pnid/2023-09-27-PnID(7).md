<input type="file" id="imageInput" accept="image/*">
<button onclick="uploadImage()">이미지 업로드</button>
<div id="preview"></div>

<script>
  function uploadImage() {
    var input = document.getElementById('imageInput');
    var preview = document.getElementById('preview');

    var file = input.files[0];
    var reader = new FileReader();

    reader.onloadend = function() {
        var img = document.createElement('img');
        img.src = reader.result;
        img.style.maxWidth = '300px'; // 이미지의 최대 너비를 지정할 수 있습니다.
        preview.appendChild(img);
        
        // 여기에서 이미지 업로드를 처리하면 됩니다.
        // 예를 들어, 이미지를 서버에 전송하거나 다른 작업을 수행할 수 있습니다.
        // 서버에 업로드할 때는 서버 측 기술 (예: Node.js, PHP, Python 등)을 사용해야 합니다.
    }

    if (file) {
        reader.readAsDataURL(file);
    } else {
        preview.innerHTML = "이미지를 선택해주세요.";
    }
  }
</script>
