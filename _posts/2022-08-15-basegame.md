bgame = [randint(1,9),randint(1,9),randint(1,9)]

    hit3 = input("세번째 숫자를 입력하세요!")
    
    while True:
        if hit1 == hit2:
            hit2 = input("두번째 숫자를 입력하세요!")

        elif hit2 == hit3:

    hit = [int(hit1), int(hit2), int(hit3)]

    for item in range(0,3):
        if bgame[0] == hit[item]:

            else:

    print("입력숫자", hit1, hit2, hit3)
    print(run_cnt, "시도횟수")

print("Congratulation")
