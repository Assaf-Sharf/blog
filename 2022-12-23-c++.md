
상수, const 제한형
```c++
#include <iostream>

using namespace std;

int main(){
	//원의 넓이를 구하는 프로그램 만들기
	//원의 넓이 = 반지름 * 반지름 * 파이

	const float PIE = 3.1415926535;

	int r = 3;
	float s = r * r * PIE;
	//1. 바뀔 필요가 없는 수
	//2. 바뀌어서는 안되는 수
	//상수

	
	cout << s << endl;
	
	return 0;
}
```


데이터형 변환
```c++
#include <iostream>

using namespace std;

int main(){
	
	int a = 3.141592;
	cout << a << endl; //----> 3으로 출력됨 int는 정수형이라서

	//강제적으로 데이터형 변환
	//typeName(a) (typeName)a
	//ex) char ch = 'M';
	//cout << (int)ch << int(ch) << endl;  ---> 77 77 이렇게 나옴

	//C++에서는
	//static_cast<typeName> 이렇게 하면 됨
	//ex) cout << static_cast<int>ch << endl; ----> 77 이렇게 나옴
	return 0;
}
    /*
	1. 특정 데이터형의 변수에 다른 데이터형의 값을 대입했을 때
	2. 수식에 데이터형을 혼합하여 사용했을 때
	3. 함수에 매개변수를 전달할 때
	이런 상환에서 자동으로 데이터형을 바꿔줌
	*/
```
